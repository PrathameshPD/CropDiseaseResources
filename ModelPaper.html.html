import random
import numpy as np
import tensorflow as tf
import os
import matplotlib.pyplot as plt
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras import layers, models
from sklearn.utils.class_weight import compute_class_weight
from tensorflow.keras.applications import ResNet50, DenseNet121, EfficientNetB0, InceptionV3, VGG16

# Set seeds for reproducibility
random.seed(0)
np.random.seed(0)
tf.random.set_seed(0)

# Directories for train, validation, and test data
base_dir = 'C:\\Users\\Panda\\Desktop\\MandyaImages'

# Image Parameters
img_size = 224
batch_size = 32

# Enhanced Image Data Generators with Additional Data Augmentation
data_gen = ImageDataGenerator(
    rescale=1./255,
    validation_split=0.2,  # Use 20% of data for validation
    rotation_range=45,
    width_shift_range=0.3,
    height_shift_range=0.3,
    shear_range=0.3,
    zoom_range=0.3,
    horizontal_flip=True,
    vertical_flip=True,  # Added vertical flip for further augmentation
    brightness_range=[0.7, 1.3],  # Expanded brightness range
    fill_mode='reflect'  # Changed fill mode to reflect
)

# Train Generator
train_generator = data_gen.flow_from_directory(
    base_dir,
    target_size=(img_size, img_size),
    batch_size=batch_size,
    subset='training',
    class_mode='categorical'
)

# Validation Generator
validation_generator = data_gen.flow_from_directory(
    base_dir,
    target_size=(img_size, img_size),
    batch_size=batch_size,
    subset='validation',
    class_mode='categorical'
)

# Compute Class Weights
class_weights = compute_class_weight(
    class_weight='balanced',
    classes=np.unique(train_generator.classes),
    y=train_generator.classes
)
class_weights = {i: class_weights[i] for i in range(len(class_weights))}

# Models to Evaluate
models_to_evaluate = {
    "ResNet50": ResNet50,
    "DenseNet121": DenseNet121,
    "EfficientNetB0": EfficientNetB0,
    "InceptionV3": InceptionV3,
    "VGG16": VGG16
}

results = {}

for model_name, base_model_class in models_to_evaluate.items():
    print(f"\nTraining {model_name}...")
    base_model = base_model_class(weights='imagenet', include_top=False, input_shape=(img_size, img_size, 3))

    # Unfreeze the top layers of the base model for fine-tuning
    for layer in base_model.layers[-20:]:  # Unfreeze more layers for fine-tuning
        layer.trainable = True

    model = models.Sequential([
        base_model,
        layers.GlobalAveragePooling2D(),  # Global pooling to reduce model complexity
        layers.Dense(512, activation='relu', kernel_regularizer=tf.keras.regularizers.l2(0.001)),
        layers.Dropout(0.5),
        layers.Dense(train_generator.num_classes, activation='softmax')
    ])

    # Compile the Model
    model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=1e-5),
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])

    # Implement a learning rate scheduler and early stopping
    lr_schedule = tf.keras.callbacks.ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=5)
    early_stopping = tf.keras.callbacks.EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)

    # Train the Model
    history = model.fit(
        train_generator,
        steps_per_epoch=train_generator.samples // batch_size,
        epochs=50,
        validation_data=validation_generator,
        validation_steps=validation_generator.samples // batch_size,
        class_weight=class_weights,
        callbacks=[lr_schedule, early_stopping]
    )

    # Evaluate the Model
    val_loss, val_accuracy = model.evaluate(validation_generator, steps=validation_generator.samples // batch_size)
    results[model_name] = val_accuracy

    # Plot training & validation accuracy values
    plt.plot(history.history['accuracy'])
    plt.plot(history.history['val_accuracy'])
    plt.title(f'{model_name} Accuracy')
    plt.ylabel('Accuracy')
    plt.xlabel('Epoch')
    plt.legend(['Train', 'Validation'], loc='upper left')
    plt.show()

    # Plot training & validation loss values
    plt.plot(history.history['loss'])
    plt.plot(history.history['val_loss'])
    plt.title(f'{model_name} Loss')
    plt.ylabel('Loss')
    plt.xlabel('Epoch')
    plt.legend(['Train', 'Validation'], loc='upper left')
    plt.show()

# Compare Results
print("\nModel Performance Comparison:")
for model_name, accuracy in results.items():
    print(f"{model_name}: Validation Accuracy = {accuracy * 100:.2f}%")

# Identify the best model
best_model = max(results, key=results.get)
print(f"\nThe best performing model is {best_model} with a validation accuracy of {results[best_model] * 100:.2f}%.")

def visualize_class_distribution(generator, title="Class Distribution"):
    """
    Visualize the distribution of classes in the dataset with labels for each bar.
    Parameters:
        generator: The data generator (e.g., train_generator or validation_generator).
        title: Title of the chart.
    """
    class_counts = np.bincount(generator.classes)
    class_labels = list(generator.class_indices.keys())

    plt.figure(figsize=(10, 6))
    bars = plt.bar(class_labels, class_counts, color='skyblue')

    # Add text labels above each bar
    for bar, count in zip(bars, class_counts):
        plt.text(
            bar.get_x() + bar.get_width() / 2,  # x-coordinate
            bar.get_height() + 2,              # y-coordinate (slightly above the bar)
            str(count),                        # The count as a string
            ha='center',                       # Horizontal alignment
            va='bottom',                       # Vertical alignment
            fontsize=10                         # Font size
        )

    plt.title(title)
    plt.xlabel('Classes')
    plt.ylabel('Number of Samples')
    plt.xticks(rotation=45)
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    plt.tight_layout()  # Adjust layout to prevent clipping
    plt.show()

# Visualize class distribution for both train and validation sets
